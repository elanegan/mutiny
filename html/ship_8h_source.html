386 void sail(Ship* localShip, Wind* localWind) {
387 double shipSpeed = getSpeed(localShip);
388 int turnLevel = getTurnLevel(localShip);
389 double windSpeed = getWindSpeed(localWind);
390 double shipAngle = getDirection(localShip);
391 double windAngle = getWindDirection(localWind);
392 double xPos = getXPosition(localShip);
393 double yPos = getYPosition(localShip);
394
395 if (getSailStat(localShip) && shipSpeed < windSpeed) {
396 shipSpeed += cos(windAngle-shipAngle);
397
398 //double approachAngle = (3*M_PI)/4 - (windAngle-shipAngle);
399 //shipAngle += (shipAngle-approachAngle)*approachAngle;
400 }
401
402 shipSpeed *= 0.95; // slow down with time, water resistance
403 shipAngle += (shipSpeed/8)*(M_PI/12)*turnLevel;
404
405 xPos += shipSpeed*sin(shipAngle);
406 yPos += shipSpeed*cos(shipAngle);
407
408 setXPosition(localShip, xPos);
409 setYPosition(localShip, yPos);
410 setXVelocity(localShip, shipSpeed*sin(shipAngle));
411 setYVelocity(localShip, shipSpeed*cos(shipAngle));
412 setDirection(localShip, shipAngle);
413 }
